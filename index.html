Horas mensais

⊞
Arraste o CSV ou clique para selecionar
Google Sheets → File → Download → CSV

A planilha deve ter as colunas MÊS, SEMANA, HORAS/MINUTOS, SISTEMA, CATEGORIA e DESCRIÇÃO.
Formatos de tempo aceitos: 1:30:00 ou 1:30.

● Conectado
Escolha o método:
Como gerar a URL (uma aba por vez):
1. Abra sua planilha no Google Sheets
2. File → Share → Publish to web
3. Em "Link", selecione a aba da planilha (ex: Fevereiro)
4. Mude o formato para Comma-separated values (.csv)
5. Clique em Publish e copie a URL gerada
Planilha com várias abas mensais? Use Apps Script abaixo — conecta uma vez e alterna o mês no relatório.
Como configurar:
1. Abra sua planilha → Extensions → Apps Script
2. Apague o código existente e cole o código abaixo
3. Deploy → New deployment → Web app
4. Execute as: Me · Who has access: Anyone
5. Clique em Deploy, autorize e copie a URL 
function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const params = (e && e.parameter) || {};

  if (params.action === 'list') {
    const sheets = ss.getSheets().map(function (s) {
      return { name: s.getName(), gid: String(s.getSheetId()) };
    });
    return jsonOut({ sheets: sheets });
  }

  var sh = null;
  if (params.sheet) {
    sh = ss.getSheetByName(params.sheet);
    if (!sh) {
      var gid = String(params.sheet);
      var all = ss.getSheets();
      for (var i = 0; i < all.length; i++) {
        if (String(all[i].getSheetId()) === gid) { sh = all[i]; break; }
      }
    }
  }
  if (!sh) sh = ss.getActiveSheet();

  const data = sh.getDataRange().getDisplayValues();
  if (!data.length) return jsonOut([]);
  const headers = data[0].map(function (h) { return String(h).trim(); });
  const rows = data.slice(1)
    .filter(function (r) { return r.some(function (c) { return c !== ''; }); })
    .map(function (r) {
      const obj = {};
      headers.forEach(function (h, i) { obj[h] = r[i]; });
      return obj;
    });
  return jsonOut(rows);
}

function jsonOut(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
